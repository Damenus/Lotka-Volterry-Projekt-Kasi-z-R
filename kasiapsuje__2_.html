=====================================================

Rwnanie Lotki-Volterry

Model drapienik-ofiara

=====================================================

Jest to nieliniowy ukad rwna rniczkowych pierwszego stopnia zaproponowany w 1926 przez Vito Volterr do opisu populacji ryb odawianych w Morzu Adriatyckim. Niezalenie od Volterry rwnowane rwnania do opisu oscylacji ste substancji w hipotetycznej reakcji chemicznej otrzyma Alfred James Lotka w 1920 roku.

Rwnanie Lotki-Voltery stanowi model ukadw dynamicznych wystpujcych w ekosystemach (np. w symulacji zachowania populacji ofiar i drapienikw).

1.

Rozwizanie dla pierwszych parametrw

od 1 do 1000 z krokiem 1

library(deSolve) #zacza bliblioteke ode
vollot <- function(Time, State, Pars) {
  with(as.list(c(State, Pars)), {
    
    
    dOfiara        <-  (rGrow * (1 - Ofiara/K) - rIng  * Drapieznik) * Ofiara
    dDrapieznik    <-  (rIng  * Ofiara *  assEff - rMort) * Drapieznik
    
    
    
    return(list(c(dOfiara, dDrapieznik)))
   })
}
#parametry
pars  <- c(rIng   = 0.2,    # tempo przyswajania(pokarmu)
           rGrow  = 1.0,    # tempo wzrostu/przyrostu zwierzyny (ofiar)
           rMort  = 0.2 ,   # wskanik meirtelnoci drapienikw
           assEff = 0.5,    # wydajno przyswajania (cz pokarmu faktycznie przetwarzana przez organizm na energie)
           K      = 10)     # pojemno rodowiska (maksymalna populacja jak mog utrzyma zasoby danego ekosystemu)

yini  <- c(Ofiara = 1, Drapieznik = 2) #stan poczatkowy
times <- seq(0, 1000, by = 1)
out   <- ode(yini, times, vollot, pars)

library(deSolve)
LVmod <- function(Time, State, Pars) {
  with(as.list(c(State, Pars)), {
   
    
    dOfiara        <-  (rGrow * (1 - Ofiara/K) - rIng  * Drapieznik) * Ofiara
    dDrapieznik    <-  (rIng  * Ofiara *  assEff - rMort) * Drapieznik
  
    
    return(list(c(dOfiara, dDrapieznik)))
   })
}

pars  <- c(rIng   = 0.2,    
           rGrow  = 1.0,    
           rMort  = 0.2 ,   
           assEff = 0.5,   
           K      = 10)     

yini  <- c(Ofiara = 1, Drapieznik = 2)
times <- seq(0, 100, by = 0.1)
out   <- ode(yini, times, LVmod, pars)
